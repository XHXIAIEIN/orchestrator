```markdown
# Data Contract

Canonical boundary between User Layer (never auto-modified) and System Layer (safe for tools to update).
All automation scripts (compiler, updaters, hooks) MUST read this contract before touching files.

## User Layer — NEVER auto-modify

These files contain user customization, personal data, or hand-tuned configuration.
Automation may READ them but MUST NOT write, overwrite, or delete.

| Path Pattern | Description |
|-------------|-------------|
| `SOUL/private/**` | Identity, relationship, voice, calibration — personal data |
| `.claude/hooks/**` | User-configured hooks |
| `.claude/settings.json` | User IDE/tool settings |
| `.env` | Environment secrets |
| `data/events.db` | Runtime data — append-only by running system |
| `config/exec-policy.yaml` | Execution policy — user-tuned |
| `departments/*/blueprint.yaml` | Department configs — user may customize |

## System Layer — Safe to auto-update

These files are generated, compiled, or maintained by tools.
Automation may overwrite them freely (they can be regenerated from source).

| Path Pattern | Description |
|-------------|-------------|
| `.claude/boot.md` | Compiled output of SOUL compiler |
| `.claude/context/*.md` | Context packs — compiled output |
| `SOUL/public/prompts/**` | System prompts — versioned, replaceable |
| `src/**` | Application code — git-tracked |
| `scripts/**` | Utility scripts — git-tracked |
| `docs/**` | Documentation — git-tracked |
| `tmp/**` | Ephemeral workspace — always safe to overwrite |

## Shared Layer — Auto-modify with constraints

These files are system-managed but contain user-visible structure.
Automation may modify them but MUST preserve existing entries (append-only or merge).

| Path Pattern | Constraint |
|-------------|-----------|
| `docs/architecture/PATTERNS.md` | Append-only: never remove existing entries |
| `SOUL/management.md` | Merge-only: preserve user edits, add new sections |

## Enforcement

1. `SOUL/tools/compiler.py` — reads this contract to validate it only writes to System Layer paths
2. Any future auto-update script — MUST load and respect this contract
3. Gate Function "Modify Core Config" in CLAUDE.md — references this contract
```

```python
import re
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parent.parent
CONTRACT_PATH = REPO_ROOT / "DATA_CONTRACT.md"


def parse_contract(path: Path = CONTRACT_PATH) -> dict[str, list[str]]:
    """Parse DATA_CONTRACT.md into {layer: [path_patterns]}."""
    text = path.read_text(encoding="utf-8")
    layers: dict[str, list[str]] = {}
    current_layer = None

    for line in text.split("\n"):
        if line.startswith("## ") and "Layer" in line:
            if "User" in line:
                current_layer = "user"
            elif "System" in line:
                current_layer = "system"
            elif "Shared" in line:
                current_layer = "shared"
            layers.setdefault(current_layer, [])
        elif current_layer and line.startswith("| `"):
            match = re.match(r"\| `([^`]+)`", line)
            if match:
                layers[current_layer].append(match.group(1))

    return layers
```

```python
def validate_output_paths():
    """Verify compiler outputs are in Data Contract's System Layer.

    Reads DATA_CONTRACT.md and checks that BOOT_PATH and CONTEXT_DIR
    fall within declared System Layer paths. Raises if not.
    """
    contract = PROJECT_ROOT / "DATA_CONTRACT.md"
    if not contract.exists():
        return  # No contract = no enforcement (graceful degradation)

    import re as _re
    text = contract.read_text(encoding="utf-8")
    system_patterns: list[str] = []
    in_system = False
    for line in text.split("\n"):
        if line.startswith("## ") and "System" in line:
            in_system = True
            continue
        if line.startswith("## "):
            in_system = False
        if in_system:
            m = _re.match(r"\| `([^`]+)`", line)
            if m:
                system_patterns.append(m.group(1))

    # Check boot.md output
    boot_rel = str(BOOT_PATH.relative_to(PROJECT_ROOT)).replace("\\", "/")
    ctx_rel = str(CONTEXT_DIR.relative_to(PROJECT_ROOT)).replace("\\", "/")

    def _in_system(path_str: str) -> bool:
        for pat in system_patterns:
            prefix = pat.rstrip("*").rstrip("/")
            if path_str.startswith(prefix):
                return True
        return False

    if not _in_system(boot_rel):
        raise RuntimeError(
            f"Compiler output {boot_rel} is NOT in Data Contract System Layer! "
            f"Update DATA_CONTRACT.md or change output path."
        )
    if not _in_system(ctx_rel):
        raise RuntimeError(
            f"Compiler output {ctx_rel} is NOT in Data Contract System Layer!"
        )
```

```python
# src/governance/audit/agent_ipc.py
"""
File-based IPC for agent intermediate results.

Steal: R46 career-ops TSV-based IPC pattern.
Adapted: JSON files in tmp/agent-output/{task-id}.json.

When a sub-agent runs a long task, it writes intermediate results to disk.
If the agent crashes, the results survive. A merge function collects all
outputs for post-processing.

This complements outcome_tracker.py (planned-vs-actual) by storing
the actual work products — not just metadata about deviation.
"""
from __future__ import annotations

import json
import logging
from dataclasses import dataclass, field, asdict
from datetime import datetime, timezone
from pathlib import Path
from typing import Optional

log = logging.getLogger(__name__)

_REPO_ROOT = Path(__file__).resolve().parent
while _REPO_ROOT != _REPO_ROOT.parent and not (
    (_REPO_ROOT / "src").is_dir() and (_REPO_ROOT / "CLAUDE.md").is_file()
):
    _REPO_ROOT = _REPO_ROOT.parent

DEFAULT_OUTPUT_DIR = _REPO_ROOT / "tmp" / "agent-output"


@dataclass
class AgentOutput:
    """One agent's intermediate or final output."""

    task_id: str
    department: str
    status: str  # running | completed | failed
    result: dict = field(default_factory=dict)
    errors: list[str] = field(default_factory=list)
    timestamp: str = ""

    def __post_init__(self):
        if not self.timestamp:
            self.timestamp = datetime.now(timezone.utc).isoformat()


def write_output(
    output: AgentOutput,
    base_dir: Optional[Path] = None,
) -> Path:
    """Write agent output to a JSON file. Overwrites if exists."""
    d = base_dir or DEFAULT_OUTPUT_DIR
    d.mkdir(parents=True, exist_ok=True)
    path = d / f"{output.task_id}.json"
    path.write_text(
        json.dumps(asdict(output), ensure_ascii=False, indent=2),
        encoding="utf-8",
    )
    return path


def read_output(
    task_id: str,
    base_dir: Optional[Path] = None,
) -> Optional[AgentOutput]:
    """Read a single agent output. Returns None if not found."""
    d = base_dir or DEFAULT_OUTPUT_DIR
    path = d / f"{task_id}.json"
    if not path.exists():
        return None
    try:
        data = json.loads(path.read_text(encoding="utf-8"))
        return AgentOutput(**data)
    except (json.JSONDecodeError, TypeError) as e:
        log.warning("Failed to read agent output %s: %s", path, e)
        return None


def merge_outputs(
    base_dir: Optional[Path] = None,
) -> list[AgentOutput]:
    """Read all agent outputs from the directory."""
    d = base_dir or DEFAULT_OUTPUT_DIR
    if not d.exists():
        return []
    results = []
    for path in sorted(d.glob("*.json")):
        try:
            data = json.loads(path.read_text(encoding="utf-8"))
            results.append(AgentOutput(**data))
        except (json.JSONDecodeError, TypeError) as e:
            log.warning("Skipping corrupt output %s: %s", path, e)
    return results


def list_outputs(
    status: Optional[str] = None,
    base_dir: Optional[Path] = None,
) -> list[AgentOutput]:
    """List outputs, optionally filtered by status."""
    all_outputs = merge_outputs(base_dir)
    if status:
        return [o for o in all_outputs if o.status == status]
    return all_outputs
```

```python
#!/usr/bin/env python3
"""
Steal system integrity checker.

Validates docs/steal/ report consistency:
1. Report numbering — gaps, duplicates
2. P0 pattern inventory — how many are declared across all reports
3. File naming convention — R<N>-<topic>-steal.md
4. Round number extraction from both filename and content

Steal: R46 career-ops Pipeline Integrity Chain pattern.
Adapted from their 5-script chain to our markdown-as-database needs.

Usage:
    python scripts/verify_steal.py              # check docs/steal/
    python scripts/verify_steal.py --fix-names  # suggest renames for non-conforming files
"""
from __future__ import annotations

import re
import sys
from dataclasses import dataclass, field
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parent.parent
STEAL_DIR = REPO_ROOT / "docs" / "steal"

# Match R<number> in filename or content
_ROUND_RE = re.compile(r"R(\d+)")
_FILENAME_RE = re.compile(r"^R(\d+)-.*\.md$")
# Match P0 table rows (| pattern | mechanism | ... |)
_P0_ROW_RE = re.compile(r"^\|(?!\s*-+\s*\|)(?!\s*Pattern\s*\|)\s*(.+?)\s*\|")


@dataclass
class IntegrityResult:
    """One check's result."""
    check: str
    passed: bool
    message: str
    details: dict = field(default_factory=dict)


def find_steal_reports(steal_dir: Path = STEAL_DIR) -> list[Path]:
    """Find all .md files that look like steal reports (have R<N> in name or content)."""
    if not steal_dir.exists():
        return []
    reports = []
    for f in sorted(steal_dir.glob("*.md")):
        # Skip consolidated/tracking files
        if "consolidated" in f.name or "ontology" in f.name or "tracker" in f.name:
            continue
        text = f.read_text(encoding="utf-8", errors="replace")
        # Must have "Steal" or "steal" in content and an R<N> reference
        if _ROUND_RE.search(f.name) or _ROUND_RE.search(text[:500]):
            reports.append(f)
    return reports


def extract_round_numbers(reports: list[Path]) -> set[int]:
    """Extract round numbers from report filenames and content headers."""
    numbers = set()
    for r in reports:
        # Try filename first
        m = _FILENAME_RE.match(r.name)
        if m:
            numbers.add(int(m.group(1)))
            continue
        # Fallback: scan first line of content for R<N>
        text = r.read_text(encoding="utf-8", errors="replace")
        for line in text.split("\n")[:5]:
            m = _ROUND_RE.search(line)
            if m:
                numbers.add(int(m.group(1)))
                break
    return numbers


def check_numbering_gaps(numbers: set[int]) -> IntegrityResult:
    """Check for gaps in round numbering."""
    if not numbers:
        return IntegrityResult(
            check="numbering_gaps",
            passed=True,
            message="No reports found",
        )
    low, high = min(numbers), max(numbers)
    expected = set(range(low, high + 1))
    missing = sorted(expected - numbers)

    if not missing:
        return IntegrityResult(
            check="numbering_gaps",
            passed=True,
            message=f"Rounds {low}-{high}: no gaps",
            details={"range": [low, high], "missing": []},
        )
    return IntegrityResult(
        check="numbering_gaps",
        passed=False,
        message=f"Rounds {low}-{high}: {len(missing)} gaps: {missing}",
        details={"range": [low, high], "missing": missing},
    )


def check_p0_patterns(reports: list[Path]) -> IntegrityResult:
    """Count P0 patterns across all reports."""
    total_p0 = 0
    by_report: dict[str, int] = {}

    for r in reports:
        text = r.read_text(encoding="utf-8", errors="replace")
        in_p0 = False
        count = 0
        for line in text.split("\n"):
            if "P0" in line and "Must Steal" in line:
                in_p0 = True
                continue
            if in_p0 and line.startswith("### "):
                in_p0 = False
            if in_p0 and _P0_ROW_RE.match(line):
                count += 1
        if count > 0:
            by_report[r.name] = count
            total_p0 += count

    return IntegrityResult(
        check="p0_inventory",
        passed=True,
        message=f"{total_p0} P0 patterns across {len(by_report)} reports",
        details={"total_p0": total_p0, "by_report": by_report},
    )


def check_naming_convention(reports: list[Path]) -> IntegrityResult:
    """Check which reports follow R<N>-<topic>-steal.md naming."""
    conforming = []
    non_conforming = []
    for r in reports:
        if _FILENAME_RE.match(r.name):
            conforming.append(r.name)
        else:
            non_conforming.append(r.name)

    passed = len(non_conforming) == 0
    return IntegrityResult(
        check="naming_convention",
        passed=passed,
        message=f"{len(conforming)} conforming, {len(non_conforming)} non-conforming",
        details={"conforming": len(conforming), "non_conforming": non_conforming},
    )


def run_all_checks(steal_dir: Path = STEAL_DIR) -> list[IntegrityResult]:
    """Run all integrity checks."""
    reports = find_steal_reports(steal_dir)
    numbers = extract_round_numbers(reports)
```
